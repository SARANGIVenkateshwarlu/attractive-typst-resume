DIRECTORIES = [
    ".github/workflows",
    ".github/ISSUE_TEMPLATE",
    "requirements",
    "src/{package_name}",
    "src/{package_name}/data",
    "src/{package_name}/models",
    "src/{package_name}/utils",
    "scripts",
    "notebooks/examples",
    "tests/integration",
    "tests/unit",
    "docs/api",
    "docs/examples",
    "docs/_static",
    "data/raw",
    "data/processed",
    "models/trained",
    "models/pretrained",
]


def create_project(project_name: str) -> None:
    root = Path(project_name)
    package_name = project_name.replace("-", "_")

    for directory in DIRECTORIES:
        path = root / directory.format(package_name=package_name)
        path.mkdir(parents=True, exist_ok=True)

    for file_path, content in FILES.items():
        path = root / file_path.format(package_name=package_name)
        path.parent.mkdir(parents=True, exist_ok=True)
        path.write_text(
            content.format(project_name=project_name, package_name=package_name),
            encoding="utf-8",
        )

    print(f"Project '{project_name}' created successfully at: {root.resolve()}")


def main():
    if len(sys.argv) != 2:
        print("Usage: python template.py <project-name>")
        sys.exit(1)

    project_name = sys.argv[1].strip()
    if not project_name:
        print("Error: project name cannot be empty.")
        sys.exit(1)

    create_project(project_name)


if __name__ == "__main__":
    main()
